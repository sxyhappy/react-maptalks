# Marker

## Example

```jsx live=true
function MarkerExample() {
  const view = { center: [-0.113049,51.49856], pitch: 0, zoom: 14 };
  const [visible, setVisible] = useState(true);
  const markerConfig = {
    'symbol' : {
      'markerFile'   : 'https://maptalks.org/examples/en/style/image-marker/raw/3.png',
      'markerWidth'  : 28,
      'markerHeight' : 40,
      'markerDx'     : 0,
      'markerDy'     : 0,
      'markerOpacity': 1
    },
    coordinate: [-0.113049,51.49856],
    cursor: 'Pointer'
  }

  const handleClick = (e) => {
    window.alert(JSON.stringify(e.target))
  }

  return (
    <div style={{height: 400}}>
      <button onClick={() => setVisible(!visible)}>change visible</button>
      <MtMap {...view}>
        <MtTileLayer
          id={"base"}
          urlTemplate="https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}.png"
          subdomains={['a','b','c','d']}
          attribution={'&copy; <a href="http://osm.org">OpenStreetMap</a> contributors, &copy; <a href="https://carto.com/">CARTO</a>'}
        />

        <MtVectorLayer id={'vector'}>
          <MtMarker {...markerConfig} visible={visible}>
            <MtInfoWindow title={'infoTitle'}><>content</></MtInfoWindow>
          </MtMarker>
        </MtVectorLayer>
      </MtMap>
    </div>
  )
}

```
