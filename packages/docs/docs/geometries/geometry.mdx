# Geometry

## Example

```jsx live=true
function GeometryExample() {
  const view = { center: [-0.113049,51.49856], pitch: 0, zoom: 14 };
  const [visible, setVisible] = useState(true);
  const profile = {
    "feature": {
      "type": "Feature",
      "id" : "point1",
      "geometry": {"type": "Point", "coordinates": [-0.113049,51.49856]},
      "properties": {"prop0": "value0"}
    },
    "options":{
      "draggable" : true
    },
    "symbol":{
      "markerFile"  : "https://maptalks.org/examples/en/style/image-marker/raw/1.png",
      'markerWidth'  : 28,
      'markerHeight' : 40,
      'markerDx'     : 0,
      'markerDy'     : 0,
      'markerOpacity': 1
    }
  }

  const handleClick = (e) => {
    window.alert(JSON.stringify(e.target))
  }

  return (
    <>
      <button onClick={() => setVisible(!visible)}>change visible</button>
      <MtMap {...view} style={{height: 400}}>
        <MtTileLayer
          id={"base"}
          urlTemplate="https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}.png"
          subdomains={['a','b','c','d']}
          attribution={'&copy; <a href="http://osm.org">OpenStreetMap</a> contributors, &copy; <a href="https://carto.com/">CARTO</a>'}
        />

        <MtVectorLayer id={'vector'}>
          <MtGeometry profile={profile} visible={visible} onClick={handleClick} />
        </MtVectorLayer>
      </MtMap>
    </>
  )
}

```
