# MtMap

## Example

```jsx live=true
function MapExample() {
  const ref = useRef();
  const [view, setView] = useState({ center: [-0.113049,51.49856], pitch: 0 });

  const handleMapReady = map => {
    console.log(map);
  }

  const handleViewChange = view => {
    setView(view.new)
  }

  useEffect(() => {
    if (ref.current) {
      console.log('ref current', ref.current)
    }
  }, [ref])

  return (
    <div style={{height: 400}}>
      <button onClick={() => setView({ ...view, center: [-0.12,51.49] })}>设置中心</button>
      <button onClick={() => setView({ ...view, pitch: 50 })}>设置倾斜</button>
      <div>{JSON.stringify(view)}</div>
      <MtMap {...view} onReady={handleMapReady} ref={ref} onViewchange={handleViewChange}>
        <MtTileLayer
          id={"base"}
          urlTemplate="https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}.png"
          subdomains={['a','b','c','d']}
          attribution={'&copy; <a href="http://osm.org">OpenStreetMap</a> contributors, &copy; <a href="https://carto.com/">CARTO</a>'}
        />
      </MtMap>
    </div>
  )
}

```
