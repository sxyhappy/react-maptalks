# MtMap

# Example
```jsx live=true
function MapExample() {
  const ref = useRef();
  const [center, setCenter] = useState([-0.113049,51.49856]);

  const handleMapReady = map => {
    console.log(map);
  }

  return (
    <div style={{height: 500}}>
      <button onClick={() => setCenter([-0.12,51.49])}>设置中心</button>
      <MtMap center={center} zoom={14} onReady={handleMapReady} ref={ref}>
        <MtTileLayer
          id={"base"}
          urlTemplate="https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}.png"
          subdomains={['a','b','c','d']}
          attribution={'&copy; <a href="http://osm.org">OpenStreetMap</a> contributors, &copy; <a href="https://carto.com/">CARTO</a>'}
        />
      </MtMap>
    </div>
  )
}

```
